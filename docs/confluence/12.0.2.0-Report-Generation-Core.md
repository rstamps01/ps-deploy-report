### **Task 2.1: Core PDF Generator Class**

#### **Implementation:** `src/reports/pdf_generator.py`

```python
from reportlab.lib.pagesizes import letter, A4
from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
from reportlab.lib.units import inch
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle, PageBreak
from reportlab.lib import colors
from reportlab.lib.enums import TA_CENTER, TA_LEFT, TA_RIGHT
from reportlab.pdfgen import canvas
from datetime import datetime
import os
from typing import Dict, List, Any

class VastPDFGenerator:
    """Professional PDF report generator for VAST As-Built reports."""
    
    def __init__(self, output_path: str, cluster_data: Dict):
        """
        Initialize PDF generator with output path and cluster data.
        
        Args:
            output_path: Full path for the output PDF file
            cluster_data: Processed cluster data from data processor
        """
        self.output_path = output_path
        self.cluster_data = cluster_data
        self.doc = SimpleDocTemplate(
            output_path,
            pagesize=A4,
            rightMargin=72,
            leftMargin=72,
            topMargin=72,
            bottomMargin=72
        )
        self.styles = self._create_custom_styles()
        self.story = []
        
    def _create_custom_styles(self) -> Dict:
        """Create custom styles for VAST branding and professional formatting."""
        styles = getSampleStyleSheet()
        
        # VAST Brand Colors
        vast_blue = colors.Color(0.0, 0.3, 0.6)  # VAST primary blue
        vast_gray = colors.Color(0.4, 0.4, 0.4)  # VAST secondary gray
        
        custom_styles = {
            'title': ParagraphStyle(
                'VastTitle',
                parent=styles['Title'],
                fontSize=24,
                textColor=vast_blue,
                spaceAfter=30,
                alignment=TA_CENTER,
                fontName='Helvetica-Bold'
            ),
            'heading1': ParagraphStyle(
                'VastHeading1',
                parent=styles['Heading1'],
                fontSize=18,
                textColor=vast_blue,
                spaceAfter=12,
                spaceBefore=20,
                fontName='Helvetica-Bold'
            ),
            'heading2': ParagraphStyle(
                'VastHeading2',
                parent=styles['Heading2'],
                fontSize=14,
                textColor=vast_blue,
                spaceAfter=8,
                spaceBefore=16,
                fontName='Helvetica-Bold'
            ),
            'normal': ParagraphStyle(
                'VastNormal',
                parent=styles['Normal'],
                fontSize=10,
                spaceAfter=6,
                fontName='Helvetica'
            ),
            'table_header': ParagraphStyle(
                'VastTableHeader',
                parent=styles['Normal'],
                fontSize=9,
                textColor=colors.white,
                fontName='Helvetica-Bold',
                alignment=TA_CENTER
            ),
            'table_cell': ParagraphStyle(
                'VastTableCell',
                parent=styles['Normal'],
                fontSize=9,
                fontName='Helvetica'
            )
        }
        
        return custom_styles
    
    def generate_report(self) -> bool:
        """
        Generate the complete PDF report.
        
        Returns:
            bool: True if successful, False otherwise
        """
        try:
            # Build report sections in order
            self._add_title_page()
            self._add_table_of_contents()
            self._add_executive_summary()
            self._add_hardware_inventory()
            self._add_network_configuration()
            self._add_data_protection()
            self._add_support_information()
            
            # Build the PDF
            self.doc.build(self.story, onFirstPage=self._add_header_footer, 
                          onLaterPages=self._add_header_footer)
            
            return True
            
        except Exception as e:
            print(f"Error generating PDF report: {e}")
            return False
    
    def _add_header_footer(self, canvas, doc):
        """Add header and footer to each page."""
        canvas.saveState()
        
        # Header
        canvas.setFont('Helvetica-Bold', 10)
        canvas.setFillColor(colors.Color(0.0, 0.3, 0.6))
        canvas.drawString(72, doc.height + 50, "VAST Data As-Built Report")
        
        # Footer
        canvas.setFont('Helvetica', 8)
        canvas.setFillColor(colors.gray)
        canvas.drawString(72, 30, f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        canvas.drawRightString(doc.width + 72, 30, f"Page {doc.page}")
        
        canvas.restoreState()
```

```python
def _add_title_page(self):
    """Create professional title page with VAST branding."""
    
    # VAST Logo placeholder (would be actual logo in production)
    logo_placeholder = Paragraph(
        "<b>VAST DATA</b>",
        ParagraphStyle(
            'Logo',
            fontSize=36,
            textColor=colors.Color(0.0, 0.3, 0.6),
            alignment=TA_CENTER,
            fontName='Helvetica-Bold'
        )
    )
    
    # Report title
    title = Paragraph("As-Built Report", self.styles['title'])
    
    # Cluster information
    cluster_name = self.cluster_data.get('cluster_overview', {}).get('cluster_name', 'Unknown')
    cluster_psnt = self.cluster_data.get('cluster_overview', {}).get('cluster_psnt', 'Unknown')
    
    subtitle = Paragraph(
        f"<b>Cluster:</b> {cluster_name}<br/>"
        f"<b>PSNT:</b> {cluster_psnt}<br/>"
        f"<b>Generated:</b> {datetime.now().strftime('%B %d, %Y')}",
        ParagraphStyle(
            'Subtitle',
            fontSize=14,
            alignment=TA_CENTER,
            spaceAfter=20
        )
    )
    
    # Add elements to story
    self.story.extend([
        Spacer(1, 2*inch),
        logo_placeholder,
        Spacer(1, 1*inch),
        title,
        Spacer(1, 0.5*inch),
        subtitle,
        PageBreak()
    ])

def _add_table_of_contents(self):
    """Generate table of contents."""
    toc_title = Paragraph("Table of Contents", self.styles['heading1'])
    
    toc_items = [
        "1. Executive Summary",
        "2. Hardware Inventory", 
        "3. Network Configuration",
        "4. Data Protection",
        "5. Support Information"
    ]
    
    self.story.append(toc_title)
    
    for item in toc_items:
        toc_item = Paragraph(item, self.styles['normal'])
        self.story.append(toc_item)
    
    self.story.append(PageBreak())
```

```python
def generate_report(self) -> bool:
    """
    Generate the complete PDF report with comprehensive error handling.
    
    Returns:
        bool: True if successful, False otherwise
    """
    try:
        # Validate input data
        if not self._validate_input_data():
            print("Error: Invalid or incomplete input data")
            return False
        
        # Build report sections in order
        self._add_title_page()
        self._add_table_of_contents()
        self._add_executive_summary()
        self._add_hardware_inventory()
        self._add_network_configuration()
        self._add_data_protection()
        self._add_support_information()
        
        # Build the PDF
        self.doc.build(self.story, onFirstPage=self._add_header_footer, 
                      onLaterPages=self._add_header_footer)
        
        print(f"PDF report generated successfully: {self.output_path}")
        return True
        
    except Exception as e:
        print(f"Error generating PDF report: {e}")
        return False

def _validate_input_data(self) -> bool:
    """Validate that required data is present."""
    
    required_sections = ['cluster_overview', 'hardware_inventory']
    
    for section in required_sections:
        if section not in self.cluster_data:
            print(f"Missing required section: {section}")
            return False
    
    # Validate cluster overview has minimum required fields
    cluster_overview = self.cluster_data['cluster_overview']
    required_fields = ['cluster_name', 'cluster_psnt']
    
    for field in required_fields:
        if field not in cluster_overview or not cluster_overview[field]:
            print(f"Missing required field in cluster_overview: {field}")
            return False
    
    return True
```
