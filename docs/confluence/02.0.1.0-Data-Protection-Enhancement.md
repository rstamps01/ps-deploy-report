```python
def _extract_retention_policies(self, snapshot_policies_data: List) -> List[Dict]:
    """Extract retention policies from snapshot policies."""
    policies = []
    for policy in snapshot_policies_data:
        retention_info = {
            'name': policy.get('name', 'Unknown'),
            'daily_retention': policy.get('daily_retention', 0),
            'weekly_retention': policy.get('weekly_retention', 0),
            'monthly_retention': policy.get('monthly_retention', 0),
            'yearly_retention': policy.get('yearly_retention', 0)
        }
        policies.append(retention_info)
    return policies
```
