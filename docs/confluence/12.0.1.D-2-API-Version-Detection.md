```python
class VastVersionDetector:
    """Detects VAST cluster and API versions automatically."""
    
    def __init__(self, client):
        self.client = client
        self.logger = logging.getLogger(__name__)
        self.version_cache = {}
    
    def detect_cluster_version(self) -> VastVersionInfo:
        """
        Detect VAST cluster version through multiple methods.
        
        Returns:
            VastVersionInfo: Comprehensive version information
        """
        try:
            # Method 1: Direct version endpoint
            version_info = self._try_version_endpoint()
            if version_info:
                return version_info
            
            # Method 2: Cluster info analysis
            version_info = self._analyze_cluster_info()
            if version_info:
                return version_info
            
            # Method 3: Feature detection
            version_info = self._detect_by_features()
            if version_info:
                return version_info
            
            # Fallback: Assume minimum supported version
            return self._get_fallback_version()
            
        except Exception as e:
            self.logger.error(f"Version detection failed: {e}")
            return self._get_fallback_version()
    
    def _try_version_endpoint(self) -> Optional[VastVersionInfo]:
        """Try to get version from dedicated endpoint."""
        endpoints_to_try = [
            '/api/version/',
            '/api/v7/version/',
            '/api/system/version/',
            '/api/cluster/version/'
        ]
        
        for endpoint in endpoints_to_try:
            try:
                response = self.client._make_request('GET', endpoint)
                if response:
                    return self._parse_version_response(response)
            except Exception:
                continue
        
        return None
    
    def _analyze_cluster_info(self) -> Optional[VastVersionInfo]:
        """Analyze cluster information to determine version."""
        try:
            cluster_data = self.client._make_request('GET', '/api/clusters/')
            if cluster_data and isinstance(cluster_data, list) and cluster_data:
                cluster = cluster_data[0]
                
                # Version indicators in cluster data
                version_indicators = {
                    'software_version': cluster.get('software_version'),
                    'api_version': cluster.get('api_version'),
                    'build_version': cluster.get('build_version'),
                    'release_version': cluster.get('release_version')
                }
                
                return self._parse_version_indicators(version_indicators)
        except Exception as e:
            self.logger.debug(f"Cluster info analysis failed: {e}")
        
        return None
    
    def _detect_by_features(self) -> Optional[VastVersionInfo]:
        """Detect version by testing for specific API features."""
        feature_tests = [
            # Test for v7 features
            {
                'version': '7.0',
                'endpoints': ['/api/v7/', '/api/quotas/', '/api/qos/'],
                'required_count': 2
            },
            # Test for v6 features
            {
                'version': '6.0',
                'endpoints': ['/api/v6/', '/api/snapshots/', '/api/replication/'],
                'required_count': 2
            }
        ]
        
        for test in feature_tests:
            available_count = 0
            for endpoint in test['endpoints']:
                try:
                    response = self.client._make_request('GET', endpoint)
                    if response is not None:
                        available_count += 1
                except Exception:
                    continue
            
            if available_count >= test['required_count']:
                return VastVersionInfo(
                    api_version=test['version'],
                    detection_method='feature_detection',
                    confidence=available_count / len(test['endpoints'])
                )
        
        return None

@dataclass
class VastVersionInfo:
    """Comprehensive version information for VAST cluster."""

    cluster_version: Optional[str] = None
    software_version: Optional[str] = None

    supported_features: List[str] = field(default_factory=list)
    deprecated_features: List[str] = field(default_factory=list)
```

```python
class CompatibilityValidator:
    """Validates API compatibility and feature availability."""
    
    def __init__(self, version_info: VastVersionInfo):
        self.version_info = version_info
        self.logger = logging.getLogger(__name__)
        self.compatibility_matrix = self._load_compatibility_matrix()
    
    def validate_compatibility(self) -> CompatibilityReport:
        """
        Validate overall compatibility with detected version.
        
        Returns:
            CompatibilityReport: Detailed compatibility analysis
        """
        report = CompatibilityReport(
            version_info=self.version_info,
            validation_timestamp=datetime.now()
        )
        
        # Validate core functionality
        report.core_compatibility = self._validate_core_features()
        
        # Validate endpoint availability
        report.endpoint_compatibility = self._validate_endpoints()
        
        # Validate feature support
        report.feature_compatibility = self._validate_features()
        
        # Generate recommendations
        report.recommendations = self._generate_recommendations(report)
        
        return report
    
    def _validate_core_features(self) -> CoreCompatibility:
        """Validate core API functionality."""
        core_endpoints = [
            '/api/clusters/',
            '/api/cboxes/',
            '/api/dboxes/',
            '/api/cnodes/',
            '/api/dnodes/'
        ]
        
        available_endpoints = []
        failed_endpoints = []
        
        for endpoint in core_endpoints:
            if self._test_endpoint_availability(endpoint):
                available_endpoints.append(endpoint)
            else:
                failed_endpoints.append(endpoint)
        
        compatibility_score = len(available_endpoints) / len(core_endpoints)
        
        return CoreCompatibility(
            score=compatibility_score,
            available_endpoints=available_endpoints,
            failed_endpoints=failed_endpoints,
            is_compatible=compatibility_score >= 0.8  # 80% threshold
        )
    
    def _validate_endpoints(self) -> Dict[str, EndpointCompatibility]:
        """Validate individual endpoint compatibility."""
        endpoint_results = {}
        
        for category, endpoints in ENDPOINT_MATRIX.items():
            category_results = []
            
            for endpoint_config in endpoints:
                compatibility = self._validate_single_endpoint(endpoint_config)
                category_results.append(compatibility)
            
            endpoint_results[category] = category_results
        
        return endpoint_results
    
    def _validate_single_endpoint(self, endpoint_config: EndpointConfig) -> EndpointCompatibility:
        """Validate a single endpoint configuration."""
        endpoint = endpoint_config.path
        min_version = endpoint_config.min_version
        max_version = endpoint_config.max_version
        
        # Check version compatibility
        version_compatible = self._is_version_in_range(
            self.version_info.api_version, min_version, max_version
        )
        
        # Test actual availability
        is_available = self._test_endpoint_availability(endpoint)
        
        # Check response structure compatibility
        structure_compatible = True
        if is_available:
            structure_compatible = self._validate_response_structure(
                endpoint, endpoint_config.expected_structure
            )
        
        return EndpointCompatibility(
            endpoint=endpoint,
            version_compatible=version_compatible,
            is_available=is_available,
            structure_compatible=structure_compatible,
            overall_compatible=version_compatible and is_available and structure_compatible,
            fallback_available=endpoint_config.fallback_endpoint is not None
        )

@dataclass
class CompatibilityReport:
    """Comprehensive compatibility report."""
    version_info: VastVersionInfo
    validation_timestamp: datetime
    core_compatibility: CoreCompatibility
    endpoint_compatibility: Dict[str, List[EndpointCompatibility]]
    feature_compatibility: FeatureCompatibility
    recommendations: List[str]
    overall_score: float = 0.0
    
    def __post_init__(self):
        """Calculate overall compatibility score."""
        scores = [
            self.core_compatibility.score,
            self._calculate_endpoint_score(),
            self.feature_compatibility.score
        ]
        self.overall_score = sum(scores) / len(scores)
```

```python
class DynamicEndpointMapper:
    """Maps API endpoints based on version compatibility."""
    
    def __init__(self, version_info: VastVersionInfo, compatibility_report: CompatibilityReport):
        self.version_info = version_info
        self.compatibility_report = compatibility_report
        self.logger = logging.getLogger(__name__)
        self.endpoint_map = self._build_endpoint_map()
    
    def get_endpoint(self, logical_name: str) -> Optional[EndpointMapping]:
        """
        Get the appropriate endpoint for a logical operation.
        
        Args:
            logical_name: Logical name of the operation (e.g., 'get_clusters')
        
        Returns:
            EndpointMapping: Endpoint configuration or None if not available
        """
        if logical_name not in self.endpoint_map:
            self.logger.warning(f"Unknown logical endpoint: {logical_name}")
            return None
        
        mapping = self.endpoint_map[logical_name]
        
        # Check if primary endpoint is available
        if mapping.primary_endpoint and self._is_endpoint_available(mapping.primary_endpoint):
            return EndpointMapping(
                logical_name=logical_name,
                actual_endpoint=mapping.primary_endpoint,
                endpoint_type='primary',
                response_transformer=mapping.primary_transformer
            )
        
        # Try fallback endpoints
        for fallback in mapping.fallback_endpoints:
            if self._is_endpoint_available(fallback.endpoint):
                return EndpointMapping(
                    logical_name=logical_name,
                    actual_endpoint=fallback.endpoint,
                    endpoint_type='fallback',
                    response_transformer=fallback.transformer,
                    limitations=fallback.limitations
                )
        
        # No available endpoint
        self.logger.error(f"No available endpoint for: {logical_name}")
        return None
    
    def _build_endpoint_map(self) -> Dict[str, LogicalEndpoint]:
        """Build the endpoint mapping based on version compatibility."""
        endpoint_map = {}
        
        # Define logical endpoints with version-specific mappings
        logical_endpoints = {
            'get_clusters': LogicalEndpoint(
                name='get_clusters',
                primary_endpoint='/api/clusters/',
                primary_transformer=self._transform_clusters_v7,
                fallback_endpoints=[
                    FallbackEndpoint(
                        endpoint='/api/v6/clusters/',
                        transformer=self._transform_clusters_v6,
                        min_version='6.0',
                        max_version='6.9'
                    )
                ]
            ),
            'get_cboxes': LogicalEndpoint(
                name='get_cboxes',
                primary_endpoint='/api/cboxes/',
                primary_transformer=self._transform_cboxes_v7,
                fallback_endpoints=[
                    FallbackEndpoint(
                        endpoint='/api/v6/cboxes/',
                        transformer=self._transform_cboxes_v6,
                        limitations=['No rack height information']
                    )
                ]
            ),
            'get_data_protection': LogicalEndpoint(
                name='get_data_protection',
                primary_endpoint='/api/protection/',
                primary_transformer=self._transform_protection_v7,
                fallback_endpoints=[
                    FallbackEndpoint(
                        endpoint='/api/snapshots/',
                        transformer=self._transform_snapshots_only,
                        limitations=['Snapshots only, no comprehensive protection data']
                    )
                ]
            ),
            'get_quotas': LogicalEndpoint(
                name='get_quotas',
                primary_endpoint='/api/quotas/',
                primary_transformer=self._transform_quotas_v7,
                fallback_endpoints=[
                    FallbackEndpoint(
                        endpoint=None,  # Not available in older versions
                        transformer=self._generate_empty_quotas,
                        limitations=['Quotas not supported in this version']
                    )
                ]
            )
        }
        
        # Filter based on compatibility
        for name, logical_endpoint in logical_endpoints.items():
            if self._is_logical_endpoint_supported(logical_endpoint):
                endpoint_map[name] = logical_endpoint
        
        return endpoint_map
    
    def _transform_clusters_v7(self, response_data: Any) -> Dict[str, Any]:
        """Transform v7 clusters response to standard format."""
        if not isinstance(response_data, list) or not response_data:
            return {}
        
        cluster = response_data[0]
        return {
            'cluster_name': cluster.get('name', 'Unknown'),
            'cluster_guid': cluster.get('guid', 'Unknown'),
            'cluster_psnt': cluster.get('psnt', 'Unknown'),
            'cluster_version': cluster.get('software_version', 'Unknown'),
            'api_version': '7.x',
            'rack_height': cluster.get('rack_height', 42)  # v7 feature
        }
    
    def _transform_clusters_v6(self, response_data: Any) -> Dict[str, Any]:
        """Transform v6 clusters response to standard format."""
        if not isinstance(response_data, list) or not response_data:
            return {}
        
        cluster = response_data[0]
        return {
            'cluster_name': cluster.get('name', 'Unknown'),
            'cluster_guid': cluster.get('guid', 'Unknown'),
            'cluster_psnt': 'Not available in v6',  # v6 limitation
            'cluster_version': cluster.get('version', 'Unknown'),
            'api_version': '6.x',
            'rack_height': 42  # Default, not available in v6
        }
```

```python
class FeatureMatrixManager:
    """Manages feature availability across VAST versions."""
    
    def __init__(self, version_info: VastVersionInfo):
        self.version_info = version_info
        self.feature_matrix = self._load_feature_matrix()
        self.logger = logging.getLogger(__name__)
    
    def is_feature_available(self, feature_name: str) -> bool:
        """Check if a feature is available in the current version."""
        feature = self.feature_matrix.get(feature_name)
        if not feature:
            return False
        
        return self._is_version_in_range(
            self.version_info.api_version,
            feature.min_version,
            feature.max_version
        )
    
    def get_available_features(self) -> List[str]:
        """Get list of all available features for current version."""
        available = []
        for feature_name, feature in self.feature_matrix.items():
            if self.is_feature_available(feature_name):
                available.append(feature_name)
        return available
    
    def get_feature_limitations(self, feature_name: str) -> List[str]:
        """Get limitations for a specific feature in current version."""
        feature = self.feature_matrix.get(feature_name)
        if not feature or not self.is_feature_available(feature_name):
            return ['Feature not available']
        
        # Version-specific limitations
        current_version = self.version_info.api_version
        for limitation in feature.version_limitations:
            if self._is_version_in_range(current_version, limitation.min_version, limitation.max_version):
                return limitation.limitations
        
        return []
```

```python
    def _detect_and_validate_version(self):
        """Detect version and validate compatibility."""
        # Initialize version detector
        self.version_detector = VastVersionDetector(self)
        
        # Detect version
        self.version_info = self.version_detector.detect_cluster_version()
        self.logger.info(f"Detected VAST version: {self.version_info}")
        
        # Validate compatibility
        self.compatibility_validator = CompatibilityValidator(self.version_info)
        self.compatibility_report = self.compatibility_validator.validate_compatibility()
        
        # Check if compatible
        if not self.compatibility_report.core_compatibility.is_compatible:
            raise VastCompatibilityError(
                f"VAST version {self.version_info.api_version} is not compatible. "
                f"Compatibility score: {self.compatibility_report.overall_score:.2f}"
            )
        
        # Initialize endpoint mapper and feature manager
        self.endpoint_mapper = DynamicEndpointMapper(self.version_info, self.compatibility_report)
        self.feature_manager = FeatureMatrixManager(self.version_info)
        
        # Log compatibility status
        self._log_compatibility_status()
    
    def _log_compatibility_status(self):
        """Log detailed compatibility status."""
        report = self.compatibility_report
        
        self.logger.info(f"Compatibility Score: {report.overall_score:.2f}")
        self.logger.info(f"Core Compatibility: {report.core_compatibility.score:.2f}")
        
        # Log available features
        available_features = self.feature_manager.get_available_features()
        self.logger.info(f"Available Features: {', '.join(available_features)}")
        
        # Log any limitations
        for recommendation in report.recommendations:
            self.logger.warning(f"Recommendation: {recommendation}")
    
    def _make_versioned_request(self, logical_endpoint: str, method: str = 'GET', **kwargs) -> Any:
        """Make a request using version-compatible endpoint mapping."""
        # Get appropriate endpoint for this version
        endpoint_mapping = self.endpoint_mapper.get_endpoint(logical_endpoint)
        
        if not endpoint_mapping:
            self.logger.warning(f"Endpoint not available: {logical_endpoint}")
            return None
        
        # Log any limitations
        if endpoint_mapping.limitations:
            self.logger.warning(f"Endpoint limitations for {logical_endpoint}: {endpoint_mapping.limitations}")
        
        # Make the request
        try:
            response = self._make_request(method, endpoint_mapping.actual_endpoint, **kwargs)
            
            # Transform response using version-specific transformer
            if response and endpoint_mapping.response_transformer:
                response = endpoint_mapping.response_transformer(response)
            
            return response
            
        except Exception as e:
            self.logger.error(f"Request failed for {logical_endpoint}: {e}")
            return None

        except Exception as e:
            self.logger.error(f"Failed to get cluster overview: {e}")
            raise VastDataError(f"Cluster overview collection failed: {str(e)}")
    
    def get_version_compatibility_report(self) -> Dict[str, Any]:
        """Get detailed version compatibility report."""
        if not self.compatibility_report:
            return {'error': 'Version detection not performed'}
        
        return {
            'version_info': asdict(self.version_info),
            'compatibility_report': asdict(self.compatibility_report),
            'available_features': self.feature_manager.get_available_features(),
            'endpoint_mappings': {
                name: {
                    'available': mapping is not None,
                    'endpoint': mapping.actual_endpoint if mapping else None,
                    'type': mapping.endpoint_type if mapping else None,
                    'limitations': mapping.limitations if mapping else []
                }
                for name in ['get_clusters', 'get_cboxes', 'get_dboxes', 'get_quotas', 'get_qos']
                for mapping in [self.endpoint_mapper.get_endpoint(name)]
            }
        }
```
